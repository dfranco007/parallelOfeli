101 void ACwithoutEdges::initialize_sums()
102 {
103  sum_in = 0;
104  sum_out = 0;
105  n_in = 0;
106  n_out = 0;
107 
108  for( int offset = 0; offset < img_size; offset++ )
109  {
110  if( phi[offset] > 0 )
111  {
112  sum_out += int(img_data[offset]);
113  n_out++;
114  }
115  else
116  {
117  sum_in += int(img_data[offset]);
118  n_in++;
119  }
120  }
121 
122  return;
123 }

125 void ACwithoutEdges::calculate_means()

137  return;
