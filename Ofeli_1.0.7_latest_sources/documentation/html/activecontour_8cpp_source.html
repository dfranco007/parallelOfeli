138 void ActiveContour::initialize_phi_with_a_shape(bool hasEllipse, double init_width, double init_height,
139  double center_x, double center_y)
140 {
141  int x, y; // position of the current pixel
142 
143  // performs an ellipse
144  if( hasEllipse )
145  {
146  for( int offset = 0; offset < img_size; offset++ )
147  {
148  find_xy_position(offset,x,y); // x and y passed by reference
149 
150  // ellipse inequation
151  if( square( double(y)-(1.0+2.0*center_y)*double(img_height)/2.0 )
152  / square( init_height*double(img_height)/2.0 )
153  + square( double(x)-(1.0+2.0*center_x)*double(img_width)/2.0 )
154  / square( init_width*double(img_width)/2.0 )
155  > 1.0
156  )
157  {
158  phi[offset] = 1; // outside boundary value
159  }
160  else
161  {
162  phi[offset] = -1; // inside boundary value
163  }
164  }
165  }
166 
167  // performs a rectangle
168  else
169  {
170  for( int offset = 0; offset < img_size; offset++ )
171  {
172  find_xy_position(offset,x,y); // x and y passed by reference
173 
174  if( double(y) > ((1.0-init_height)*double(img_height)/2.0 + center_y*double(img_height))
175  && double(y) < ((double(img_height)-(1.0-init_height)*double(img_height)/2.0) + center_y*double(img_height))
176  && double(x) > ((1.0-init_width)*double(img_width)/2.0 + center_x*double(img_width))
177  && double(x) < ((double(img_width)-(1.0-init_width)*double(img_width)/2.0) + center_x*double(img_width))
178  )
179  {
180  phi[offset] = -1; // inside boundary value
181  }
182  else
183  {
184  phi[offset] = 1; // outside boundary value
185  }
186  }
187  }
188 
189  return;
190 }

192 void ActiveContour::initialize_lists()
193 {
194  // each point of a list must be connected at least by one point of the other list
195  // eliminate redundant points in phi if needed, and initialize Lout and Lin
196  // so you can pass to the constructor phi_init1, a binarized buffer
197  // with 1 for outside region and -1 for inside region to simplify your task
198  for( int offset = 0; offset < img_size; offset++ )
199  {
200  if( phi[offset] == 1 ) // outside boundary value
201  {
202  if( isRedundantLoutPoint(offset) )
203  {
204  phi[offset] = 3; // exterior value
205  }
206  else
207  {
208  Lout.push_front(offset);
209  }
210  }
211 
212  if( phi[offset] == -1 ) // inside boundary value
213  {
214  if( isRedundantLinPoint(offset) )
215  {
216  phi[offset] = -3; // interior value
217  }
218  else
219  {
220  Lin.push_front(offset);
221  }
222  }
223  }
224 
225  return;
226 }

254 const int* const ActiveContour::make_gaussian_kernel(int kernel_length1, double sigma1)
255 {
256  // kernel_length impair and strictly positive
257  if( kernel_length1 % 2 == 0 )
258  {
259  kernel_length1--;
260  }
261 
262  if( kernel_length1 < 1 )
263  {
264  kernel_length1 = 1;
265  }
266 
267  // protection against /0
268  if( sigma1 < 0.000000001 )
269  {
270  sigma1 = 0.000000001;
271  }
272 
273  int x, y; // position of the current pixel
274 
275  const int kernel_size1 = kernel_length1*kernel_length1;
276  const int kernel_radius1 = (kernel_length1-1)/2;
277 
278  int* const gaussian_kernel1 = new int[kernel_size1];
279 
280  for( int offset = 0; offset < kernel_size1; offset++ )
281  {
282  // offset = x+y*kernel_length so
283  y = offset/kernel_length1;
284  x = offset-y*kernel_length1;
285 
286  gaussian_kernel1[offset] = int( 0.5 +
287  100000.0
288  * std::exp( -( double( square(y-kernel_radius1)
289  + square(x-kernel_radius1) )
290  ) / (2.0*square(sigma1)) ) );
291  }
292 
293  return gaussian_kernel1;
294 }

302 void ActiveContour::do_one_iteration_in_cycle1()
303 {
304  // means of the Chan-Vese model for children classes ACwithoutEdges and ACwithoutEdgesYUV
305  calculate_means(); // virtual function for region-based models
306 
307 
308  hasOutwardEvolution = false;
309 
310  for( ofeli::list<int>::iterator Lout_point = Lout.begin(); !Lout_point.end(); )
311  {
312  if( compute_external_speed_Fd(*Lout_point) > 0 )
313  {
314  hasOutwardEvolution = true;
315 
316  // updates of the variables to calculate the means Cout and Cin
317  updates_for_means_in1(); // virtual function for region-based models
318 
319  Lout_point = switch_in(Lout_point); // outward local movement
320  // switch_in function returns a new Lout_point
321  // which is the next point of the former Lout_point
322  }
323  else
324  {
325  ++Lout_point;
326  }
327  }
328 
329 
330  clean_Lin(); // eliminate Lin redundant points
331 
332 
333  hasInwardEvolution = false;
334 
335  for( ofeli::list<int>::iterator Lin_point = Lin.begin(); !Lin_point.end(); )
336  {
337  if( compute_external_speed_Fd(*Lin_point) < 0 )
338  {
339  hasInwardEvolution = true;
340 
341  // updates of the variables to calculate the means Cout and Cin
342  updates_for_means_out1(); // virtual function for region-based models
343 
344  Lin_point = switch_out(Lin_point); // inward local movement
345  // switch_out function returns a new Lin_point
346  // which is the next point of the former Lin_point
347  }
348  else
349  {
350  ++Lin_point;
351  }
352  }
353 
354 
355  clean_Lout(); // eliminate Lout redundant points
356 
357 
358  iteration++;
359 
360  return;
361 }

425 ActiveContour& ActiveContour::operator++()
426 {
427 
428  // Fast Two Cycle algorithm
429 
430  while( !isStopped )
431  {
432 
434  while( Na < Na_max && !hasLastCycle2 )
435  {
436  do_one_iteration_in_cycle1();
437  calculate_stopping_conditions1(); // it computes hasLastCycle2
438  Na++;
439  return *this; // just one iteration is performed
440  }
442 
443  if( hasCycle2 )
444  {
446  while( Ns < Ns_max )
447  {
448  do_one_iteration_in_cycle2();
449  Ns++;
450  return *this; // just one iteration is performed
451  }
453 
454  if( hasLastCycle2 ) // a last cycle 2 has been performed before
455  {
456  isStopped = true;
457  }
458  else
459  {
460  calculate_stopping_conditions2(); // it computes isStopped
461  }
462  }
463  else
464  {
465  if( hasLastCycle2 )
466  {
467  isStopped = true;
468  }
469  }
470 
471  Na = 0;
472  Ns = 0;
473  }
474 
475  return *this;
476 }

563 ofeli::list<int>::iterator ActiveContour::switch_in(ofeli::list<int>::iterator Lout_point)
564 {
565  int offset, x, y;
566  offset = *Lout_point;
567  find_xy_position(offset,x,y); // x and y passed by reference
568 
569  // Outward local movement
570 
571  #ifndef VERSION_8_CONNECTED_NEIGHBORHOOD
572 
573  //========== 4-connected neighborhood =========
574  if( y > 0 )
575  {
576  add_Rout_neighbor_to_Lout( find_offset(x,y-1) );
577  }
578  if( x > 0 )
579  {
580  add_Rout_neighbor_to_Lout( find_offset(x-1,y) );
581  }
582  if( x < img_width-1 )
583  {
584  add_Rout_neighbor_to_Lout( find_offset(x+1,y) );
585  }
586  if( y < img_height-1 )
587  {
588  add_Rout_neighbor_to_Lout( find_offset(x,y+1) );
589  }
590  //=================================================
591 
592  #else
593 
594  //========== 8-connected neighborhood =========
595  // if not in the image's border, no neighbors' tests
596  if( x > 0 && x < img_width-1 && y > 0 && y < img_height-1 )
597  {
598  // scan through a 8-connected neighborhood
599  for( int dy = -1; dy <= 1; dy++ )
600  {
601  for( int dx = -1; dx <= 1; dx++ )
602  {
603  if( !( dx == 0 && dy == 0 ) )
604  {
605  add_Rout_neighbor_to_Lout( find_offset(x+dx,y+dy) );
606  }
607  }
608  }
609  }
610  // if in the border, neighbors' tests
611  else
612  {
613  // scan through a 8-connected neighborhood
614  for( int dy = -1; dy <= 1; dy++ )
615  {
616  for( int dx = -1; dx <= 1; dx++ )
617  {
618  if( !( dx == 0 && dy == 0 ) )
619  {
620  // existence tests
621  if( x+dx >= 0 && x+dx < img_width && y+dy >= 0 && y+dy < img_height )
622  {
623  add_Rout_neighbor_to_Lout( find_offset(x+dx,y+dy) );
624  }
625  }
626  }
627  }
628  }
629  //=================================================
630 
631  #endif
632 
633  phi[offset] = -1; // 1 ==> -1
634  return Lin.splice_front(Lout_point); // Lout_point ∈ Lout ==> Lout_point ∈ Lin
635  // return a new Lout_point which is the next point of the former Lout_point
636  // obviously, this new Lout_point ∈ Lout
637 }
