104 void ACwithoutEdgesYUV::initialize_for_each_frame()
105 {
106  int it = get_iteration();
107 
108  ActiveContour::initialize_for_each_frame();
109 
110  // condition for video tracking :
111  // If the active contour moves enough ( >120 ), variables sums, n_in, n_out
112  // are too different and are recalculated.
113  // If the active contour does not evolve ( < 5 because a last cycle 2 is performed )
114  // in the case of uniform images (or during the object initialization),
115  // variables are (re)calculated for convergence.
116  // Otherwise, variables are not calculated.
117  if( it > 120 || it < 5 )
118  {
119  initialize_sums();
120  }
121  calculate_means();
122 
123  return;
124 }

126 void ACwithoutEdgesYUV::initialize_sums()
127 {
128  sum_in_R = 0;
129  sum_out_R = 0;
130 
131  sum_in_G = 0;
132  sum_out_G = 0;
133 
134  sum_in_B = 0;
135  sum_out_B = 0;
136 
137  n_in = 0;
138  n_out = 0;
139 
140  for( int offset = 0; offset < img_size; offset++ )
141  {
142  RGB_from_buffer(offset);
143 
144  if( phi[offset] > 0 )
145  {
146  sum_out_R += R;
147  sum_out_G += G;
148  sum_out_B += B;
149 
150  n_out++;
151  }
152  else
153  {
154  sum_in_R += R;
155  sum_in_G += G;
156  sum_in_B += B;
157 
158  n_in++;
159  }
160  }
161 
162  return;
163 }

165 void ACwithoutEdgesYUV::calculate_means()
166 {
167  if( n_out != 0 )
168  {
169  CoutR = sum_out_R/n_out;
170  CoutG = sum_out_G/n_out;
171  CoutB = sum_out_B/n_out;
172  }
173 
174  if( n_in != 0 )
175  {
176  CinR = sum_in_R/n_in;
177  CinG = sum_in_G/n_in;
178  CinB = sum_in_B/n_in;
179  }
180 
181  calculate_YUV(CoutR, CoutG, CoutB, CoutYUV);
182  calculate_YUV(CinR, CinG, CinB, CinYUV);
183 
184  return;
185 }

187 int ACwithoutEdgesYUV::compute_external_speed_Fd(int offset)
188 {
189  RGB_from_buffer(offset);
190 
191  int YUV[3];
192  calculate_YUV(R, G, B, YUV);
193 
194  return lambda_out*( alpha*square(YUV[0]-CoutYUV[0])
195  + beta*square(YUV[1]-CoutYUV[1])
196  + gamma*square(YUV[2]-CoutYUV[2]) )
197 
198  - lambda_in*( alpha*square(YUV[0]-CinYUV[0])
199  + beta*square(YUV[1]-CinYUV[1])
200  + gamma*square(YUV[2]-CinYUV[2]) );
201 }
