48 HausdorffDistance::HausdorffDistance(const int* list1_1, int list1_1_length, int img1_1_width, int img1_1_height,
49  const int* list2_1, int list2_1_length, int img2_1_width, int img2_1_height) :
50  list1(list1_1), list1_length(list1_1_length), img1_width(img1_1_width), img1_height(img1_1_height),
51  list2(list2_1), list2_length(list2_1_length), img2_width(img2_1_width), img2_height(img2_1_height),
52  max_dist( calculate_max_dist(img1_1_width,img1_1_height,img2_1_width,img2_1_height) )
53 {
54  if( list1_1 == NULL )
55  {
56  std::cerr << std::endl << "The pointer list1_1 must be a non-null pointer, it must be allocated." << std::endl;
57  }
58  if( list1_1_length == 0 )
59  {
60  std::cerr << std::endl << "list1_1 must not be empty." << std::endl;
61  }
62 
63  if( list2_1 == NULL )
64  {
65  std::cerr << std::endl << "The pointer list2_1 must be a non-null pointer, it must be allocated." << std::endl;
66  }
67  if( list2_1_length == 0 )
68  {
69  std::cerr << std::endl << "list2_1 must not be empty." << std::endl;
70  }

74  int x1, y1;
75  int sum_x1 = 0;
76  int sum_y1 = 0;
77 
78  for( int index = 0; index < list1_length; index++ )
79  {
80  y1 = list1[index]/img1_width;
81  x1 = list1[index]-y1*img1_width;
82 
83  sum_x1 += x1;
84  sum_y1 += y1;
85  }
86 
87  // centroid of list 1
88  if( list1_length != 0 )
89  {

91  mean_y1 = double(sum_y1)/double(list1_length);
92  }

95  int sum_x2 = 0;
96  int sum_y2 = 0;
97 
98  for( int index = 0; index < list2_length; index++ )
99  {

101  x2 = list2[index]-y2*img2_width;
102 
103  sum_x2 += x2;
104  sum_y2 += y2;
105  }
106 
107  // centroid of list 2
108  if( list2_length != 0 )
109  {
110  mean_x2 = double(sum_x2)/double(list2_length);
111  mean_y2 = double(sum_y2)/double(list2_length);
112  }

133 void HausdorffDistance::compute_hausdorff_dist(double& hausdorff_dist, double& modified_hausdorff_dist) const
134 {
135  int x1, y1, x2, y2;
136  double relative_x1, relative_y1, relative_x2, relative_y2;
137  double euclidean_dist, sum_min_dist, min_dist;
138 
139  sum_min_dist = 0.0;
140 
141  // initialization in order to maximize
142  hausdorff_dist = 0.0;
143 
144  // for each point of list 1
145  for( int index1 = 0; index1 < list1_length; index1++ )
146  {
147  y1 = list1[index1]/img1_width;
148  x1 = list1[index1]-y1*img1_width;
149 
150  // position in the centroid frame
151  relative_x1 = double(x1)-mean_x1;
152  relative_y1 = double(y1)-mean_y1;
153 
154  // initialization in order to minimize
155  min_dist = 999999.999;
156 
157  // computes the distance of all points of list 2
158  for( int index2 = 0; index2 < list2_length; index2++ )
159  {
160  y2 = list2[index2]/img2_width;
161  x2 = list2[index2]-y2*img2_width;
162 
163  // position in the centroid frame
164  relative_x2 = double(x2)-mean_x2;
165  relative_y2 = double(y2)-mean_y2;
166 
167 
168  // computes the euclidean distance
169  euclidean_dist = std::sqrt( square(relative_x2-relative_x1) + square(relative_y2-relative_y1) );
170 
171  // it minimizes min_dist
172  if( euclidean_dist < min_dist )
173  {
174  min_dist = euclidean_dist;
175  }
176  }
177 
178  // it maximizes hausdorff_dist
179  if( min_dist > hausdorff_dist )
180  {
181  hausdorff_dist = min_dist;
182  }
183 
184  sum_min_dist += min_dist;
185  }
186 
187  // modified hausdorff distance = mean of min_dist
188  modified_hausdorff_dist = sum_min_dist/double(list1_length);
189 
190  return;
191 }

206 {
